#include <bits/stdc++.h>
using ll = long long;

const ll MAXN = 2e5 + 5;
const ll MAXD = 30;  
int n, q;
int parent[MAXN][MAXD];

int jump(int a, int d) {
	for (int i = 0; i < MAXD; i++)
		if (d & (1 << i)) a = parent[a][i];
	return a;
}

int main() {
	std::cin >> n >> q;
	for (int i = 1; i <= n; i++) { cin >> parent[i][0]; }
	
	for (int d = 1; d < MAXD; d++)
		for (int i = 1; i <= n; i++) { parent[i][d] = parent[parent[i][d - 1]][d - 1]; }
	while (q--) {
		int a, d;
		std::cin >> a >> d;
		std::cout << jump(a, d) << '\n';
	}
}
